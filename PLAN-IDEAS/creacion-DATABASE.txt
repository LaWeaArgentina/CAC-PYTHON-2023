-- Creación de la tabla para Cliente
CREATE TABLE Cliente (
    idCliente INT PRIMARY KEY,
    nombre VARCHAR(255),
    direccion VARCHAR(255),
    telefono VARCHAR(20),
    email VARCHAR(255)
);

-- Creación de la tabla para Presupuesto
CREATE TABLE Presupuesto (
    idPresupuesto INT PRIMARY KEY,
    nombre VARCHAR(255),
    fecha DATE,
    total DECIMAL,
    idCliente INT,
    FOREIGN KEY (idCliente) REFERENCES Cliente(idCliente)
);

-- Creación de la tabla para Proyecto
CREATE TABLE Proyecto (
    idProyecto INT PRIMARY KEY,
    nombre VARCHAR(255),
    descripcion TEXT,
    fechaInicio DATE,
    fechaFin DATE,
    idPresupuesto INT,
    FOREIGN KEY (idPresupuesto) REFERENCES Presupuesto(idPresupuesto)
);

-- Creación de la tabla para Item
CREATE TABLE Item (
    idItem INT PRIMARY KEY,
    nombre VARCHAR(255),
    costo DECIMAL,
    idPresupuesto INT,
    FOREIGN KEY (idPresupuesto) REFERENCES Presupuesto(idPresupuesto)
);
